---
Write a JavaScript function that returns the factorial of a number using a "while" loop.
---

```initial
  let result = -1;
```

```initial
  let result = 0;
```

```initial
  let result = 2;
```

```js
function factorial(n) {
  let i = 1;
  while (i <= n) {
    result *= i;
    i++;
  }
  return result;
}
```

```js
function factorial(n) {
  while (n > 1) {
    result *= n;
    n--;
  }
  return result;
}
```

```js
function factorial(n) {
  while (n > 0) {
    result *= n;
    n--;
  }
  return result;
}
```
